pfx:

$cert = New-SelfSignedCertificate -Type CodeSigningCert -Subject "CN=Neptune" -CertStoreLocation Cert:\CurrentUser\My -NotAfter (Get-Date).AddYears(5)

$password = ConvertTo-SecureString -String "Neptune@2026" -Force -AsPlainText

Export-PfxCertificate -Cert $cert -FilePath "E:\neptune\neptune.pfx" -Password $password

certutil -urlcache -split -f http://192.168.21.242:8082/neptune.pfx  neptune.pfx

certutil -p Neptune@2026 -importpfx Root neptune.pfx

certutil -store Root | findstr Neptune

signtool sign /tr http://timestamp.digicert.com /td SHA256 /fd SHA256 /f "E:\neptune\neptune.pfx" /p Neptune@2026 "E:\neptune\build\neptune.exe"

signtool verify /pa /v neptune.exe
